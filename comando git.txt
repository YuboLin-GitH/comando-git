# 初始化本地仓库（只需一次）
git init  # 初始化当前目录为 git 仓库

# 配置用户名和邮箱（只需一次）
git config --global user.name "你的名字"
git config --global user.email "你的邮箱@example.com"

# 设置中文注释支持（推荐）
git config --global i18n.commitEncoding utf-8
git config --global i18n.logOutputEncoding utf-8
git config --global core.quotepath false

# 查看当前仓库状态
git status  # 显示当前变更、未提交文件

# 添加所有文件到暂存区
git add .  # 添加当前目录下所有文件

# 提交到本地仓库（带中文注释）
git commit -m "这里是你的中文提交说明"

# 关联远程仓库（只需一次）
git remote add origin https://github.com/你的用户名/你的仓库名.git

# 推送到远程仓库（首次推送建议加-u参数）
git push -u origin master  # 推送到 master 分支
# 或者
git push -u origin main    # 推送到 main 分支

# 拉取远程仓库最新内容并合并
git pull origin master  # 拉取 master 分支
# 或
git pull origin main    # 拉取 main 分支

# 查看提交历史
git log  # 显示所有提交记录

# 删除远程仓库所有内容（慎用！）
git rm -r *  # 删除所有文件
git commit -m "清空仓库"
git push

# 克隆远程仓库到本地
git clone https://github.com/你的用户名/你的仓库名.git

# 查看分支
git branch  # 显示本地分支
git branch -r  # 显示远程分支

# 切换分支
git checkout 分支名

# 合并分支
git merge 分支名

# 删除本地分支
git branch -d 分支名

# 删除远程分支
git push origin --delete 分支名
